Firestore Security Rules

এই rules Firebase Console → Firestore → Rules-এ দিন:

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ── Helper: admin check ──
    function isAdmin() {
      return request.auth != null
        && request.auth.token.email in ['roni84211@gmail.com'];
    }
    function isLoggedIn() {
      return request.auth != null;
    }

    // ✅ Files — সবাই read
    //    Admin: full write
    //    Authenticated user: শুধু adUnlockCount ও reviews update করতে পারবে
    match /files/{fileId} {
      allow read: if true;
      allow create, delete: if isAdmin();
      allow update: if isAdmin()
        || (
          isLoggedIn()
          && request.resource.data.diff(resource.data).affectedKeys()
              .hasOnly(['adUnlockCount', 'reviews', 'downloads'])
        );
    }
    // ✅ Sliders — সবাই read, শুধু admin write
    match /sliders/{id} {
      allow read: if true;
      allow write: if isAdmin();
    }
    // ✅ Categories — সবাই read, শুধু admin write
    match /categories/{id} {
      allow read: if true;
      allow write: if isAdmin();
    }
    // ✅ Sponsor Posts — সবাই read, শুধু admin write
    match /sponsorPosts/{id} {
      allow read: if true;
      allow write: if isAdmin();
    }
    // ✅ Settings — সবাই read (app settings দরকার), শুধু admin write
    match /settings/{id} {
      allow read: if true;
      allow write: if isAdmin();
    }
    // ✅ broadcastHistory — authenticated users read, শুধু admin write
    match /broadcastHistory/{id} {
      allow read: if isLoggedIn();
      allow write: if isAdmin();
    }
    // ✅ Users — নিজের data নিজে read/write, admin সব read করতে পারবে
    match /users/{userId} {
      allow read: if isLoggedIn()
        && (request.auth.uid == userId || isAdmin());
      allow write: if isLoggedIn()
        && request.auth.uid == userId;
    }
  }
}

⚠️ Realtime Database (RTDB) Rules — এটাও দিন:

{
  "rules": {
    "presence": {
      ".read": true,
      ".write": true
    },
    "users": {
      ".read": "auth != null",
      "$uid": {
        ".write": true
      }
    }
  }
}

Firebase Console → Realtime Database → Rules এ এটা দিলে Admin panel users দেখতে পাবে এবং User panel থেকে data লেখা যাবে।
